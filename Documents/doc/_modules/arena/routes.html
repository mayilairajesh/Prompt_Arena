def send_otp_email(email, otp):
    """
    Send a one-time password (OTP) to the specified email address.

    Uses Gmail SMTP (or any SMTP server defined in Config) to deliver a plain+HTML email
    containing the 6-digit OTP. Used during login/signup for secure access.

    Args:
        email (str): Recipient's email address.
        otp (str): The OTP code to include in the message.

    Returns:
        bool: True if email was sent successfully, False otherwise.

    Example:
        >>> send_otp_email("user@example.com", "123456")
        True
    """
    msg = MIMEMultipart("alternative")
    msg["Subject"] = "Your OTP for PromptArena"
    msg["From"] = Config.MAIL_DEFAULT_SENDER
    msg["To"] = email

    text = f"Your OTP is: {otp}"
    html = f"""
    <html>
      <body>
        <h3>Welcome to <strong>PromptArena</strong>!</h3>
        <p>Your one-time password (OTP) is: <strong>{otp}</strong></p>
        <p>This code expires in 10 minutes.</p>
      </body>
    </html>
    """

    part1 = MIMEText(text, "plain")
    part2 = MIMEText(html, "html")

    msg.attach(part1)
    msg.attach(part2)

    try:
        with smtplib.SMTP(Config.MAIL_SERVER, Config.MAIL_PORT) as server:
            server.starttls()
            server.login(Config.MAIL_USERNAME, Config.MAIL_PASSWORD)
            server.sendmail(msg["From"], msg["To"], msg.as_string())
        return True
    except Exception as e:
        print(f"Error sending email: {e}")
        return False

@main.route('/login', methods=['GET', 'POST'])
def login():
    """
    Handle user login via email and OTP.

    On GET: Display the login form.
    On POST:
    - Validate required fields (first name, last name, email).
    - Create a new user if email doesn't exist.
    - Generate a TOTP-based OTP and store it in the database.
    - Send OTP via email using `send_otp_email`.

    Returns:
        str or werkzeug.wrappers.response.Response: Rendered login template or redirect to OTP verification.
    """
    if request.method == 'POST':
        email = request.form.get('email').strip().lower()
        first_name = request.form.get('first_name').strip()
        last_name = request.form.get('last_name').strip()
        mobile_number = request.form.get('mobile_number', '').strip()

        # Validate required fields
        if not first_name or not last_name or not email:
            flash("All fields marked with * are required.", "error")
            return render_template('login.html')

        # Check if user exists
        user = User.query.filter_by(email=email).first()

        if not user:
            # Create new user
            user = User(
                first_name=first_name,
                last_name=last_name,
                email=email,
                mobile_number=mobile_number
            )
            db.session.add(user)
            db.session.commit()

        # Generate OTP
        totp = TOTP(Config.OTP_SECRET)
        otp = totp.now()
        user.otp = otp
        db.session.commit()

        # Send OTP
        if send_otp_email(user.email, otp):
            flash("OTP has been sent to your email.", "info")
            return redirect(url_for('main.verify_otp', email=user.email))
        else:
            flash("Failed to send OTP. Please try again.", "error")

    return render_template('login.html')


@main.route('/verify-otp/<email>', methods=['GET', 'POST'])
def verify_otp(email):
    """
    Verify the OTP entered by the user.

    On POST: Validates the OTP using TOTP (valid within 30 seconds window).
    On success: Clears OTP, logs in the user, and redirects to welcome.

    Args:
        email (str): URL parameter â€” the user's email address.

    Returns:
        str or werkzeug.wrappers.response.Response: OTP form or redirect on success/failure.
    """
    user = User.query.filter_by(email=email).first()
    if not user or not user.otp:
        flash("Invalid or expired session.", "error")
        return redirect(url_for('main.login'))

    if request.method == 'POST':
        entered_otp = request.form.get('otp')
        totp = TOTP(Config.OTP_SECRET)

        if totp.verify(entered_otp):  # Validates within 30s window
            user.otp = None  # Clear OTP after use
            db.session.commit()
            login_user(user)
            return redirect(url_for('main.welcome'))
        else:
            flash("Invalid OTP. Please try again.", "error")

    return render_template('verify_otp.html', email=email)


@main.route('/login/gmail')
def gmail_login():
    """
    Initiate Google OAuth2 login flow.

    Fetches Google's OpenID configuration and redirects the user to Google's
    authorization endpoint with proper scope and redirect URI.

    Returns:
        werkzeug.wrappers.response.Response: Redirect to Google's login page.
    """
    google_discovery_url = "https://accounts.google.com/.well-known/openid-configuration"
    config = requests.get(google_discovery_url).json()

    auth_params = {
        'client_id': Config.GOOGLE_CLIENT_ID,
        'redirect_uri': url_for('main.gmail_callback', _external=True),
        'response_type': 'code',
        'scope': 'openid email profile',
        'prompt': 'select_account'
    }

    auth_url = f"{config['authorization_endpoint']}?{urlencode(auth_params)}"
    return redirect(auth_url)


@main.route('/login/gmail/callback')
def gmail_callback():
    """
    Handle Google OAuth2 callback after user authorization.

    Exchanges the authorization code for an access token, then fetches user info
    (email, name). Creates or updates the local user account and logs them in.

    Returns:
        werkzeug.wrappers.response.Response: Redirect to welcome or login on failure.
    """
    code = request.args.get('code')

    # Exchange code for token
    google_discovery_url = "https://accounts.google.com/.well-known/openid-configuration"
    config = requests.get(google_discovery_url).json()

    token_endpoint = config["token_endpoint"]
    data = {
        'client_id': Config.GOOGLE_CLIENT_ID,
        'client_secret': Config.GOOGLE_CLIENT_SECRET,
        'code': code,
        'grant_type': 'authorization_code',
        'redirect_uri': url_for('main.gmail_callback', _external=True)
    }

    token_response = requests.post(token_endpoint, data=data)
    if token_response.status_code != 200:
        flash("Authentication failed with Gmail.", "error")
        return redirect(url_for('main.login'))

    token_json = token_response.json()
    userinfo_url = config["userinfo_endpoint"]
    userinfo_response = requests.get(
        userinfo_url,
        headers={'Authorization': f"Bearer {token_json['access_token']}"}
    )

    if userinfo_response.status_code != 200:
        flash("Could not retrieve user info.", "error")
        return redirect(url_for('main.login'))

    userinfo = userinfo_response.json()

    email = userinfo['email']
    first_name = userinfo.get('given_name', 'User')
    last_name = userinfo.get('family_name', '')
    mobile_number = ''  # Google doesn't provide mobile via basic profile

    # Get or create user
    user = User.query.filter_by(email=email).first()
    if not user:
        user = User(
            first_name=first_name,
            last_name=last_name,
            email=email,
            mobile_number=mobile_number
        )
        db.session.add(user)
        db.session.commit()
    else:
        # Update name if needed
        user.first_name = first_name
        user.last_name = last_name
        db.session.commit()

    login_user(user)
    return redirect(url_for('main.welcome'))
