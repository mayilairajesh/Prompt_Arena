"""
Database Models for Prompt Arena

This module defines the SQLAlchemy data models used in the Prompt Arena application.
It includes:
- User: Handles authentication, profile data, OTP login, and guest/full account status.
- PromptSubmission: Stores user-submitted prompts and AI-generated evaluations.

The models use Flask-SQLAlchemy and integrate with Flask-Login via UserMixin.
"""

from flask_login import UserMixin
from werkzeug.security import generate_password_hash, check_password_hash
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

# Initialize SQLAlchemy instance (to be bound in app.py)
db = SQLAlchemy()

class User(db.Model, UserMixin):
    """
    Represents a user in the Prompt Arena system.

    This model supports both guest logins (via email + OTP) and full registered users.
    Guests can later upgrade to full accounts by setting a password and completing their profile.

    Attributes:
        id (int): Primary key.
        first_name (str): User's first name.
        last_name (str): User's last name.
        email (str): Unique email address; used for login and OTP delivery.
        mobile_number (str, optional): Contact number.
        password_hash (str): Hashed password (None for guests until set).
        otp (str): One-time password for email-based login (6 digits).
        otp_created_at (datetime): Timestamp when OTP was generated.
        is_guest (bool): True if user hasn't completed profile; False otherwise.
        created_at (datetime): Account creation time.
        updated_at (datetime): Last update time (auto-updated).
    """

    __tablename__ = 'user'

    id = db.Column(db.Integer, primary_key=True)
    first_name = db.Column(db.String(100), nullable=False)
    last_name = db.Column(db.String(100), nullable=False)
    email = db.Column(db.String(120), unique=True, nullable=False, index=True)
    mobile_number = db.Column(db.String(20))
    password_hash = db.Column(db.String(256))
    otp = db.Column(db.String(6))
    otp_created_at = db.Column(db.DateTime)
    is_guest = db.Column(db.Boolean, default=True, nullable=False)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    def check_password(self, password):
        """
        Verify a plaintext password against the stored hash.

        Args:
            password (str): Password to check.

        Returns:
            bool: True if the password matches, False otherwise or if no hash exists.
        """
        return check_password_hash(self.password_hash, password) if self.password_hash else False



    def __repr__(self):
        """
        Return a readable string representation of the User.

        Returns:
            str: Example: "<User Alice Smith>"
        """
        return f"<User {self.first_name} {self.last_name}>"

class PromptSubmission(db.Model):
    """
    Represents a single prompt submitted by a user in the battle arena.

    Each submission is evaluated by an AI judge and assigned scores for creativity,
    relevance, clarity, and an overall score, along with feedback.

    Attributes:
        id (int): Primary key.
        user_id (int): Foreign key referencing the User who submitted.
        level (str): Difficulty level (e.g., 'Basic', 'Intermediate', 'Pro').
        battle_type (str): Type of battle (e.g., '1v1', 'Free For All').
        prompt_text (str): The actual prompt submitted by the user.
        creativity_score (float): Score from 0–10 for originality.
        relevance_score (float): Score from 0–10 for topic alignment.
        clarity_score (float): Score from 0–10 for coherence and structure.
        overall_score (float): Average of the three scores.
        feedback (str): Constructive AI-generated feedback.
        timestamp (datetime): When the prompt was submitted.
        user (User): SQLAlchemy relationship to the User model.
    """

    __tablename__ = 'prompt_submission'

    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False, index=True)
    level = db.Column(db.String(20), nullable=False)
    battle_type = db.Column(db.String(30), nullable=False)
    prompt_text = db.Column(db.Text, nullable=False)
    creativity_score = db.Column(db.Float, nullable=False)
    relevance_score = db.Column(db.Float, nullable=False)
    clarity_score = db.Column(db.Float, nullable=False)
    overall_score = db.Column(db.Float, nullable=False)
    feedback = db.Column(db.Text, nullable=False)
    timestamp = db.Column(db.DateTime, default=datetime.utcnow)

    # Relationship to User
    user = db.relationship('User', backref=db.backref('submissions', lazy=True))

    def __repr__(self):
        """
        Return a readable string representation of the PromptSubmission.

        Returns:
            str: Example: "<PromptSubmission 8.5/10>"
        """
        return f"<PromptSubmission {self.overall_score}/10>"
